DECLARE
    n_sales NUMBER := 300000;
    n_commission NUMBER(10, 2) := 0;
BEGIN
    IF n_sales > 200000 THEN
        n_commission := n_sales * 0.1; -- Set commission to 10%
    ELSE
        n_commission := n_sales * 0.05; -- Set commission to 5%
    END IF;
    -- Additional statements (if needed)
    DBMS_OUTPUT.PUT_LINE('Commission: ' || n_commission);
END;
/
