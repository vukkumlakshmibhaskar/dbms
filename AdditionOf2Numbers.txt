Using Variables:

DECLARE
    x NUMBER(5);
    y NUMBER(5);
    z NUMBER(7);
BEGIN
    x := 10; -- Assigning 10 to x
    y := 20; -- Assigning 20 to y
    z := x + y; -- Sum of x and y
    DBMS_OUTPUT.PUT_LINE('Sum is ' || z);
END;
/




Using a Function (Runtime Input):


CREATE OR REPLACE FUNCTION add_two_nums(num1 NUMERIC, num2 NUMERIC)
    RETURN NUMERIC AS
BEGIN
    RETURN num1 + num2;
END;
$$ LANGUAGE PLPGSQL;


SELECT add_two_nums(5, 6); ///output
